# Pre-requisites
- Install tmux 
    brew install tmux
- Install tpm
    git clone https://github.com/tmux-plugins/tpm ~/.tmux/plugins/tpm

# Install plugins 
prefix + shift + i

# Commands
- tmux ls
- tmux attach -t session_name
- tmux detach
- prefix + c: creates a new window
- prefix + number: go to  window
- prefix + ,: rename window
- prefix + w: see all windows + sessions
